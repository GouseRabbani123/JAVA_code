# How to take an input in Java

#Code start here
--> code to take input as a number

import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        //A Simple Hello World
  
  Scanner scn = new Scanner(System.in);// it takes the input.
  int n = scn.nextInt();// here this "scn.nextInt" command is used for getting the integer value. 
   for(int i =0; i<=n;i++){
       System.out.println(i);
   }     
    }
}

    }
